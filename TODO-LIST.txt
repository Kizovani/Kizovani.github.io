    <script>
        const openCircle = document.getElementById('open-circle');
        const closeCircle = document.getElementById('close-circle');
        const menu = document.getElementById('menu');

        gsap.set(closeCircle, { scale: 0.5 });

        function showMenu() {
            gsap.to(openCircle, { duration: 0.3, y: 60, opacity: 0 });
            gsap.to(menu, { duration: 0.3, display: 'block', opacity: 1, y: 0 });
        }

        function hideMenu() {
            gsap.to(menu, { duration: 0.3, display: 'none', opacity: 0, y: 60 });
            gsap.to(openCircle, { duration: 0.3, y: 0, opacity: 1 });
        }

        openCircle.addEventListener('mouseover', () => {
            gsap.to(openCircle, { duration: 0.2, scale: 1.2 });
        });

        openCircle.addEventListener('mouseout', () => {
            gsap.to(openCircle, { duration: 0.2, scale: 1 });
        });

        openCircle.addEventListener('click', showMenu);
        closeCircle.addEventListener('click', hideMenu);

        closeCircle.addEventListener('mouseover', () => {
            gsap.to(closeCircle, { duration: 0.2, scale: 0.6 });
        });

        closeCircle.addEventListener('mouseout', () => {
            gsap.to(closeCircle, { duration: 0.2, scale: 0.5 });
        });
    </script>
